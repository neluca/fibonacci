\def\spiral#1{
	\pgfmathparse{int(#1)}
	\ifnum\pgfmathresult > 0
	\draw [black!20] (0, 0) rectangle (1, 1);
	\begin{scope}[shift={(1,1)}, rotate=-90, scale=1/1.6180339887]
		\spiral{#1-1}
	\end{scope}
	\draw [black!80,very thick] (0, 0) arc (180:90:1);
	\fi
}
